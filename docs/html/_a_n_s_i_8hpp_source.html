923  class colorize
924  {
925  private:

927  std::string str;
928 
929  public:
935  colorize(std::string str) : str(str)
936  {
937  fq1 = 2 * M_PI / str.size();
938  fq2 = 2 * M_PI / str.size();
939  fq3 = 2 * M_PI / str.size();
940  amp = 127;
941  cen = 128;
942  ps1 = 0 * M_PI/3;
943  ps2 = 2 * M_PI/3;
944  ps3 = 4 * M_PI/3;
945  }

954  TEMPLATE
955  friend OSTREAM& operator<<(OSTREAM& os, const colorize &obj)
956  {
957  int r = 0, g = 0, b = 0, v = 0;
958 
959  for(std::string::size_type i = 0; i < obj.str.size(); ++i)
960  {
961  if(!std::isspace(obj.str[i]))
962  {
963  r = std::sin(obj.fq1 * v + obj.ps1) * obj.amp + obj.cen;
964  g = std::sin(obj.fq2 * v + obj.ps2) * obj.amp + obj.cen;
965  b = std::sin(obj.fq3 * v + obj.ps3) * obj.amp + obj.cen;
966  v = v + 1;
967  os << fg_color(r, g, b) << obj.str[i];
968  }
969  else
970  {
971  os << obj.str[i];
972  }
973  }
974 
975  return os << reset;
976  }
