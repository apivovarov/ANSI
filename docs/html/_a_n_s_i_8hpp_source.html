31 int ANSI__init()
32 {
33 HANDLE StdOut = GetStdHandle(STD_OUTPUT_HANDLE);
34
35 if (StdOut != INVALID_HANDLE_VALUE)
36 {
37 DWORD mode = 0;
38 if (GetConsoleMode(StdOut, &mode))
39 {
40 mode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
41 if (SetConsoleMode(StdOut, mode))
42 {
43 return 0;
44 }
45 }
46 }
47
48 return GetLastError();
49 }
